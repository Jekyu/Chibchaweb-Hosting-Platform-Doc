\subsection{Requisitos del Hardware}
\begin{itemize}
    \item {\textbf{RAM:} 4GB mínimo, 16 GB ideal.}
    \item { \textbf{CPU}: 2 o más núcleos.}
    \item {\textbf{Almacenamiento}: 4 GB  mínimo, 10 GB ideal.}
\end{itemize}
